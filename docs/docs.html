Installation

Install AckerJS via npm, yarn, or pnpm:

# Using npm
npm install ackerjs

# Using yarn
yarn add ackerjs

# Using pnpm
pnpm add ackerjs

Basic Usage

AckerJS can be imported in multiple ways depending on your needs:

Import Everything

import { DOM, HTTP, Format } from 'ackerjs';

// Use the modules
const element = DOM.select('#app');
const data = await HTTP.fetchJSON('/api/data');
const formatted = Format.formatDate(new Date());

Import Specific Modules (Better Tree-Shaking)

import * as DOM from 'ackerjs/dom';
import * as HTTP from 'ackerjs/http';
import * as Format from 'ackerjs/format';

Import Individual Functions

import { select, create, append } from 'ackerjs';

const container = select('#container');
const button = create('button', { textContent: 'Click Me' });
append(container, button);

DOM Module

The DOM module provides utilities for selecting and manipulating DOM elements with type safety.

select(selector)

Select a single DOM element using a CSS selector.

select(selector: string): HTMLElement | null
selector
CSS selector string (e.g., '#id', '.class', 'div')
Returns
The first matching HTMLElement or null if not found
import { DOM } from 'ackerjs';

// Select by ID
const app = DOM.select('#app');

// Select by class
const header = DOM.select('.header');

// Select by tag
const firstDiv = DOM.select('div');

// Complex selector
const button = DOM.select('nav .menu button.active');

selectAll(selector)

Select all DOM elements matching a CSS selector.

selectAll(selector: string): HTMLElement[]
selector
CSS selector string
Returns
Array of matching HTMLElements (empty array if none found)
import { DOM } from 'ackerjs';

// Select all elements with class
const items = DOM.selectAll('.item');

// Iterate over elements
items.forEach(item => {
  DOM.addClass(item, 'processed');
});

// Get count
console.log(`Found ${items.length} items`);

create(tag, props?)

Create a new DOM element with optional properties.

create(tag: string, props?: Record<string, any>): HTMLElement
tag
HTML tag name (e.g., 'div', 'button', 'span')
props
Optional object containing element properties (className, id, textContent, etc.)
Returns
Newly created HTMLElement
import { DOM } from 'ackerjs';

// Simple element
const div = DOM.create('div');

// With properties
const button = DOM.create('button', {
  className: 'btn btn-primary',
  id: 'submit-btn',
  textContent: 'Submit',
  disabled: false
});

// With data attributes
const card = DOM.create('div', {
  className: 'card',
  'data-id': '123',
  'data-type': 'product'
});

// With inline styles
const box = DOM.create('div', {
  style: 'width: 100px; height: 100px; background: blue;'
});

append(parent, child)

Append a child element to a parent element.

append(parent: HTMLElement, child: HTMLElement): void
parent
Parent HTMLElement to append to
child
Child HTMLElement to append
import { DOM } from 'ackerjs';

const container = DOM.select('#container');
const title = DOM.create('h1', { textContent: 'Welcome' });
const paragraph = DOM.create('p', { textContent: 'Hello World' });

DOM.append(container, title);
DOM.append(container, paragraph);

remove(element)

Remove an element from the DOM.

remove(element: HTMLElement): void 
import { DOM } from 'ackerjs';

const element = DOM.select('#old-element');
DOM.remove(element);

Class Manipulation

addClass(element, ...classes)

Add one or more CSS classes to an element.

addClass(element: HTMLElement, ...classes: string[]): void 
import { DOM } from 'ackerjs';

const button = DOM.select('#btn');

// Add single class
DOM.addClass(button, 'active');

// Add multiple classes
DOM.addClass(button, 'btn', 'btn-primary', 'large');

removeClass(element, ...classes)

Remove one or more CSS classes from an element.

removeClass(element: HTMLElement, ...classes: string[]): void 
import { DOM } from 'ackerjs';

const button = DOM.select('#btn');
DOM.removeClass(button, 'active', 'loading');

toggleClass(element, className)

Toggle a CSS class on an element.

toggleClass(element: HTMLElement, className: string): void 
import { DOM } from 'ackerjs';

const menu = DOM.select('#menu');
const toggle = DOM.select('#toggle-btn');

DOM.on(toggle, 'click', () => {
  DOM.toggleClass(menu, 'open');
});

hasClass(element, className)

Check if an element has a specific CSS class.

hasClass(element: HTMLElement, className: string): boolean 
import { DOM } from 'ackerjs';

const button = DOM.select('#btn');

if (DOM.hasClass(button, 'active')) {
  console.log('Button is active');
}

Event Handling

on(element, event, handler)

Add an event listener to an element.

on(element: HTMLElement, event: string, handler: EventListener): void 
import { DOM } from 'ackerjs';

const button = DOM.select('#btn');

DOM.on(button, 'click', (e) => {
  console.log('Button clicked!', e);
});

DOM.on(button, 'mouseenter', () => {
  DOM.addClass(button, 'hover');
});

DOM.on(button, 'mouseleave', () => {
  DOM.removeClass(button, 'hover');
});

off(element, event, handler)

Remove an event listener from an element.

off(element: HTMLElement, event: string, handler: EventListener): void 
import { DOM } from 'ackerjs';

const button = DOM.select('#btn');

const clickHandler = (e) => {
  console.log('Clicked!');
};

DOM.on(button, 'click', clickHandler);

// Later, remove the listener
DOM.off(button, 'click', clickHandler);

Attributes & Properties

attr(element, name, value?)

Get or set an element's attribute.

attr(element: HTMLElement, name: string, value?: string): string | void 
import { DOM } from 'ackerjs';

const input = DOM.select('#email');

// Set attribute
DOM.attr(input, 'placeholder', 'Enter email');
DOM.attr(input, 'required', 'true');

// Get attribute
const placeholder = DOM.attr(input, 'placeholder');
console.log(placeholder); // "Enter email"

data(element, key, value?)

Get or set data attributes.

data(element: HTMLElement, key: string, value?: string): string | void 
import { DOM } from 'ackerjs';

const card = DOM.select('.card');

// Set data attribute
DOM.data(card, 'id', '123');
DOM.data(card, 'type', 'product');

// Get data attribute
const id = DOM.data(card, 'id');
console.log(id); // "123"

HTTP Module

The HTTP module provides fetch wrappers with automatic JSON parsing and error handling.

fetchJSON(url, options?)

Fetch JSON data with automatic error handling and parsing.

fetchJSON<T>(url: string, options?: RequestInit): Promise<T>
url
URL to fetch from
options
Optional fetch options (headers, credentials, etc.)
Returns
Promise resolving to parsed JSON data
import { HTTP } from 'ackerjs';

// Simple GET request
const user = await HTTP.fetchJSON('https://api.example.com/user/1');
console.log(user.name);

// With TypeScript types
interface User {
  id: number;
  name: string;
  email: string;
}

const user = await HTTP.fetchJSON<User>('/api/user/1');

// With custom headers
const data = await HTTP.fetchJSON('/api/data', {
  headers: {
    'Authorization': 'Bearer token123',
    'X-Custom-Header': 'value'
  }
});

postJSON(url, data, options?)

Send a POST request with JSON data.

postJSON<T>(url: string, data: any, options?: RequestInit): Promise<T> 
import { HTTP } from 'ackerjs';

// Create a new user
const newUser = await HTTP.postJSON('/api/users', {
  name: 'John Doe',
  email: 'john@example.com',
  age: 30
});

console.log(`Created user with ID: ${newUser.id}`);

putJSON(url, data, options?)

Send a PUT request with JSON data.

putJSON<T>(url: string, data: any, options?: RequestInit): Promise<T> 
import { HTTP } from 'ackerjs';

// Update existing user
const updated = await HTTP.putJSON('/api/users/1', {
  name: 'Jane Doe',
  email: 'jane@example.com'
});

deleteJSON(url, options?)

Send a DELETE request.

deleteJSON<T>(url: string, options?: RequestInit): Promise<T> 
import { HTTP } from 'ackerjs';

// Delete a user
await HTTP.deleteJSON('/api/users/1');
console.log('User deleted');

API Instance

Create a reusable API instance with a base URL.

createAPI(baseURL, defaultOptions?)

createAPI(baseURL: string, defaultOptions?: RequestInit): APIInstance 
import { HTTP } from 'ackerjs';

// Create API instance
const api = HTTP.createAPI('https://api.example.com', {
  headers: {
    'Authorization': 'Bearer token123',
    'Content-Type': 'application/json'
  }
});

// Use the instance
const users = await api.get('/users');
const newUser = await api.post('/users', { name: 'John' });
const updated = await api.put('/users/1', { name: 'Jane' });
await api.delete('/users/1');

Format Module

The Format module provides utilities for formatting dates, numbers, and strings.

Date Formatting

formatDate(date, locale?)

Format a date using the browser's locale or specified locale.

formatDate(date: Date, locale?: string): string 
import { Format } from 'ackerjs';

const now = new Date();

// Default locale
Format.formatDate(now);
// "11/16/2025"

// Specific locale
Format.formatDate(now, 'en-GB');
// "16/11/2025"

Format.formatDate(now, 'de-DE');
// "16.11.2025"

formatDateTime(date, locale?)

Format a date and time.

formatDateTime(date: Date, locale?: string): string 
import { Format } from 'ackerjs';

Format.formatDateTime(new Date());
// "11/16/2025, 2:30:45 PM"

formatTime(date, locale?)

Format just the time portion.

formatTime(date: Date, locale?: string): string 
import { Format } from 'ackerjs';

Format.formatTime(new Date());
// "2:30:45 PM"

Number Formatting

formatNumber(num, decimals?, locale?)

Format a number with optional decimal places.

formatNumber(num: number, decimals?: number, locale?: string): string 
import { Format } from 'ackerjs';

Format.formatNumber(1234567.89);
// "1,234,567.89"

Format.formatNumber(1234.5678, 2);
// "1,234.57"

Format.formatNumber(1234.56, 2, 'de-DE');
// "1.234,56"

formatCurrency(amount, currency?, locale?)

Format a number as currency.

formatCurrency(amount: number, currency?: string, locale?: string): string 
import { Format } from 'ackerjs';

Format.formatCurrency(1234.56);
// "$1,234.56"

Format.formatCurrency(1234.56, 'EUR', 'de-DE');
// "1.234,56 €"

Format.formatCurrency(1234.56, 'GBP', 'en-GB');
// "£1,234.56"

formatPercent(num, decimals?)

Format a number as a percentage.

formatPercent(num: number, decimals?: number): string 
import { Format } from 'ackerjs';

Format.formatPercent(0.1234);
// "12.34%"

Format.formatPercent(0.1234, 1);
// "12.3%"

String Formatting

capitalize(str)

Capitalize the first letter of a string.

capitalize(str: string): string 
import { Format } from 'ackerjs';

Format.capitalize('hello world');
// "Hello world"

capitalizeWords(str)

Capitalize the first letter of each word.

capitalizeWords(str: string): string 
import { Format } from 'ackerjs';

Format.capitalizeWords('hello world from ackerjs');
// "Hello World From Ackerjs"

truncate(str, length, suffix?)

Truncate a string to a specified length.

truncate(str: string, length: number, suffix?: string): string 
import { Format } from 'ackerjs';

Format.truncate('Hello World', 5);
// "Hello..."

Format.truncate('Hello World', 5, '…');
// "Hello…"

Format.truncate('Hi', 10);
// "Hi" (no truncation needed)

slugify(str)

Convert a string to a URL-friendly slug.

slugify(str: string): string 
import { Format } from 'ackerjs';

Format.slugify('Hello World!');
// "hello-world"

Format.slugify('AckerJS Framework 2024');
// "ackerjs-framework-2024"

Format.slugify('  Spaces   and   Punctuation!!! ');
// "spaces-and-punctuation"

Case Conversion

camelCase(str)

Convert a string to camelCase.

camelCase(str: string): string 
import { Format } from 'ackerjs';

Format.camelCase('hello-world');
// "helloWorld"

Format.camelCase('hello_world');
// "helloWorld"

Format.camelCase('Hello World');
// "helloWorld"

kebabCase(str)

Convert a string to kebab-case.

kebabCase(str: string): string 
import { Format } from 'ackerjs';

Format.kebabCase('helloWorld');
// "hello-world"

Format.kebabCase('HelloWorld');
// "hello-world"

Format.kebabCase('hello_world');
// "hello-world"

snakeCase(str)

Convert a string to snake_case.

snakeCase(str: string): string 
import { Format } from 'ackerjs';

Format.snakeCase('helloWorld');
// "hello_world"

Format.snakeCase('Hello World');
// "hello_world"

Additional Resources

TypeScript Support: AckerJS is built with TypeScript and includes full type definitions. You'll get IntelliSense and type checking out of the box.
Tree-Shaking: Import only what you need for smaller bundle sizes. Modern bundlers will automatically remove unused code.
Browser Support: AckerJS targets ES2020+ and works in all modern browsers. For older browsers, use a transpiler like Babel.
